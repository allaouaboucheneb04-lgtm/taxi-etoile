rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function signedIn() {
      return request.auth != null;
    }

    function isAdmin() {
      return signedIn() &&
        exists(/databases/$(database)/documents/admins/$(request.auth.uid)) &&
        get(/databases/$(database)/documents/admins/$(request.auth.uid)).data.active == true;
    }

    function isDriver() {
      return signedIn() &&
        exists(/databases/$(database)/documents/drivers/$(request.auth.uid)) &&
        get(/databases/$(database)/documents/drivers/$(request.auth.uid)).data.active != false;
    }

    match /reservations/{reservationId} {
      allow create: if true;

      allow read: if isAdmin() || (isDriver() && resource.data.driverId == request.auth.uid);

      allow update: if isAdmin()
        || (isDriver()
          && resource.data.driverId == request.auth.uid
          && request.resource.data.driverId == resource.data.driverId
          && request.resource.data.driverName == resource.data.driverName);

      allow delete: if isAdmin();
    }

    match /admins/{adminId} {
      allow read: if signedIn() && request.auth.uid == adminId;
      allow write: if false;
    }

    match /drivers/{driverId} {
      allow read: if isAdmin() || (signedIn() && request.auth.uid == driverId);
      allow create: if isAdmin();
      allow update: if isAdmin() || (signedIn() && request.auth.uid == driverId);
      allow delete: if isAdmin();
    }
  }
}
